41  void compute() {
42  GaussSquare<Constants::maximum_quadrature_degree> GS;
43  auto Q = GS[deg_];
44  SurfacePoint qp;
45  auto super_space = ansatz_space_.get_superspace();
46  auto element_tree = super_space.get_mesh().get_element_tree();
47  auto number_of_elements = element_tree.get_number_of_elements();
48  auto polynomial_degree = super_space.get_polynomial_degree();
49  auto polynomial_degree_plus_one_squared =
50  (polynomial_degree + 1) * (polynomial_degree + 1);
51  const auto function_space_dimension =
52  getFunctionSpaceVectorDimension<LinearOperatorTraits<LinOp>::Form>();
53  Eigen::Matrix<typename LinearFormTraits<Derived>::Scalar, Eigen::Dynamic,
54  function_space_dimension>
55  intval(polynomial_degree_plus_one_squared, function_space_dimension);
56  Eigen::Matrix<typename LinearFormTraits<Derived>::Scalar, Eigen::Dynamic,
57  function_space_dimension>
58  disc_lf_matrix(polynomial_degree_plus_one_squared * number_of_elements,
59  function_space_dimension);
60  disc_lf_matrix.setZero();
61  for (auto element = element_tree.cpbegin(); element != element_tree.cpend();
62  ++element) {
63  intval.setZero();
64  for (auto i = 0; i < Q.w_.size(); ++i) {
65  super_space.map2surface(*element, Q.xi_.col(i),
66  element->get_h() * Q.w_(i), &qp);
67  lf_.evaluateIntegrand_impl(super_space, qp, &intval);
68  }
69  disc_lf_matrix.block(polynomial_degree_plus_one_squared * element->id_, 0,
70  polynomial_degree_plus_one_squared,
71  function_space_dimension) = intval;
72  }
73  disc_lf_ =
74  ansatz_space_.get_transformation_matrix().transpose() *
75  Eigen::Map<Eigen::Matrix<typename LinearOperatorTraits<LinOp>::Scalar,
76  Eigen::Dynamic, 1>>(
77  disc_lf_matrix.data(),
78  disc_lf_matrix.rows() * disc_lf_matrix.cols());
79  return;
80  }

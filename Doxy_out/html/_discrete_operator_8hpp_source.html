24 template <typename Derived, typename Scalar>
25 struct DiscreteOperatorComputer<
26  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>, Derived> {
27  DiscreteOperatorComputer(){};
28  static void compute(
29  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> *disc_op,
30  const Derived &lin_op, const AnsatzSpace<Derived> &ansatz_space) {
31  GaussSquare<Constants::maximum_quadrature_degree> GS;
32  auto super_space = ansatz_space.get_superspace();
33  auto element_tree = super_space.get_mesh().get_element_tree();
34  auto number_of_elements = element_tree.get_number_of_elements();
35  const auto vector_dimension =
36  getFunctionSpaceVectorDimension<LinearOperatorTraits<Derived>::Form>();
37  auto polynomial_degree = super_space.get_polynomial_degree();
38  auto polynomial_degree_plus_one_squared =
39  (polynomial_degree + 1) * (polynomial_degree + 1);
40  auto ffield_deg = lin_op.get_FarfieldQuadratureDegree(polynomial_degree);
41  auto ffield_qnodes =
42  DuffyTrick::computeFfieldQnodes(super_space, GS[ffield_deg]);
43  disc_op->resize(vector_dimension * polynomial_degree_plus_one_squared *
44  number_of_elements,
45  vector_dimension * polynomial_degree_plus_one_squared *
46  number_of_elements);
47 #pragma omp parallel
48  {
49 #pragma omp single
50  {
51  for (auto element1 = element_tree.cpbegin();
52  element1 != element_tree.cpend(); ++element1)
53  for (auto element2 = element_tree.cpbegin();
54  element2 != element_tree.cpend(); ++element2) {
55 #pragma omp task firstprivate(element1, element2)
56  {
57  Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic> intval(
58  vector_dimension * polynomial_degree_plus_one_squared,
59  vector_dimension * polynomial_degree_plus_one_squared);
60  DuffyTrick::evaluateBilinearForm(lin_op, super_space, *element1,
61  *element2, GS, ffield_qnodes,
62  &intval);
63  for (auto i = 0; i < vector_dimension; ++i)
64  for (auto j = 0; j < vector_dimension; ++j)
65  disc_op->block(polynomial_degree_plus_one_squared *
66  (i * number_of_elements + element1->id_),
67  polynomial_degree_plus_one_squared *
68  (j * number_of_elements + element2->id_),
69  polynomial_degree_plus_one_squared,
70  polynomial_degree_plus_one_squared) =
71  intval.block(i * polynomial_degree_plus_one_squared,
72  j * polynomial_degree_plus_one_squared,
73  polynomial_degree_plus_one_squared,
74  polynomial_degree_plus_one_squared);
75  }
76  }
77  }
78  }
79  auto projector = ansatz_space.get_transformation_matrix();
80  disc_op[0] = projector.transpose() * (disc_op[0] * projector);
81  return;
82  }
83 };

36 struct FunctionEvaluatorEval<Scalar, DifferentialForm::DivConforming, LinOp> {

43  const Eigen::Matrix<
44  Scalar, Eigen::Dynamic,
45  getFunctionSpaceVectorDimension<DifferentialForm::DivConforming>()>
46  &coeff) const {

48  auto x_f_dx = p.segment<3>(6);
49  auto x_f_dy = p.segment<3>(9);
50  Eigen::Matrix<typename LinearOperatorTraits<LinOp>::Scalar, Eigen::Dynamic,
51  1>
52  tangential_coefficients = coeff.transpose() * super_space.basis(s);
53  return x_f_dx * tangential_coefficients(0) +
54  x_f_dy * tangential_coefficients(1);

57  Scalar evalDiv(
58  const SuperSpace<LinOp> &super_space,
59  const int polynomial_degree_plus_one_squared,
60  const ElementTreeNode &element, const SurfacePoint &p,
61  const Eigen::Matrix<
62  Scalar, Eigen::Dynamic,
63  getFunctionSpaceVectorDimension<DifferentialForm::DivConforming>()>
64  &coeff) const {
65  auto s = p.segment<2>(0);
66  auto h = element.get_h();
67  Eigen::Matrix<typename LinearOperatorTraits<LinOp>::Scalar, Eigen::Dynamic,
68  1>
69  phiPhiVec_dx = super_space.basisDx(s);
70  Eigen::Matrix<typename LinearOperatorTraits<LinOp>::Scalar, Eigen::Dynamic,
71  1>
72  phiPhiVec_dy = super_space.basisDy(s);
73  return (phiPhiVec_dx.dot(coeff.col(0)) + phiPhiVec_dy.dot(coeff.col(1))) /
74  h;
75  };

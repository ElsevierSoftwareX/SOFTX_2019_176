158  void addScaledVectorBasisInteraction(
159  Eigen::Matrix<typename LinearOperatorTraits<Derived>::Scalar,
160  Eigen::Dynamic, Eigen::Dynamic>* intval,
161  typename LinearOperatorTraits<Derived>::Scalar w,
162  const Eigen::Vector2d& s, const Eigen::Vector2d& t,
163  const Eigen::Vector3d x_f_dx, const Eigen::Vector3d x_f_dy,
164  const Eigen::Vector3d y_f_dx, const Eigen::Vector3d y_f_dy) const {
165  auto basis_interaction = basisInteraction(s, t);
166  intval->block(0, 0, polynomial_degree_plus_one_squared,
167  polynomial_degree_plus_one_squared) +=
168  w * x_f_dx.dot(y_f_dx) * basis_interaction;
169  intval->block(0, polynomial_degree_plus_one_squared,
170  polynomial_degree_plus_one_squared,
171  polynomial_degree_plus_one_squared) +=
172  w * x_f_dx.dot(y_f_dy) * basis_interaction;
173  intval->block(polynomial_degree_plus_one_squared, 0,
174  polynomial_degree_plus_one_squared,
175  polynomial_degree_plus_one_squared) +=
176  w * x_f_dy.dot(y_f_dx) * basis_interaction;
177  intval->block(polynomial_degree_plus_one_squared,
178  polynomial_degree_plus_one_squared,
179  polynomial_degree_plus_one_squared,
180  polynomial_degree_plus_one_squared) +=
181  w * x_f_dy.dot(y_f_dy) * basis_interaction;
182  }

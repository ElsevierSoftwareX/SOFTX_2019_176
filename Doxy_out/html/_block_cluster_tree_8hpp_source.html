116  template <typename Derived>
117  void init_BlockClusterTree(const LinearOperatorBase<Derived> &linOp,
118  const AnsatzSpace<Derived> &ansatz_space) {
119  if (parameters_ == nullptr) set_parameters();
120  // get element tree from ansatz space
121  auto element_tree =
122  ansatz_space.get_superspace().get_mesh().get_element_tree();
123  cluster1_ = std::addressof(element_tree.root());
124  cluster2_ = std::addressof(element_tree.root());
125  // get memory structure from element tree
126  auto mem = cluster1_->memory_;
127  // set parameters for matrix assembly
128  parameters_->max_level_ = mem->max_level_;
129  parameters_->polynomial_degree_ =
130  ansatz_space.get_superspace().get_polynomial_degree();
131  parameters_->polynomial_degree_plus_one_squared_ =
132  (parameters_->polynomial_degree_ + 1) *
133  (parameters_->polynomial_degree_ + 1);
134  parameters_->ffield_deg_ =
135  linOp.get_FarfieldQuadratureDegree(parameters_->polynomial_degree_);
136  // set up leaf_pointers
137  leaf_pointers_ = std::make_shared<std::vector<BlockClusterTree *>>();
138  leaf_pointers_->clear();
139  // block cluster tree assembly
140  rows_ = std::distance(mem->cluster_begin(*cluster1_),
141  mem->cluster_end(*cluster1_));
142  cols_ = std::distance(mem->cluster_begin(*cluster2_),
143  mem->cluster_end(*cluster2_));
144  // we let appendSubtree handle everything, since the root always
145  // returns 0 in compare cluster
146  appendSubtree(linOp, ansatz_space, cluster1_, cluster2_);
147  updateLeafPointers();
148  return;
149  }
153  template <typename Derived>
154  void appendSubtree(const LinearOperatorBase<Derived> &linear_operator,
155  const AnsatzSpace<Derived> &ansatz_space,
156  const ElementTreeNode *cluster1,
157  const ElementTreeNode *cluster2) {
158  auto mem = cluster1->memory_;
159  cc_ = compareCluster(*cluster1, *cluster2);
160  // there are children to handle
161  if (cc_ == BlockClusterAdmissibility::Refine) {
162  // reserve memory for sons_
163  sons_.resize(cluster1->sons_.size(), cluster2->sons_.size());
164  for (auto j = 0; j < sons_.cols(); ++j)
165  for (auto i = 0; i < sons_.rows(); ++i) {
166  const ElementTreeNode &son1 = mem->son(*cluster1, i);
167  const ElementTreeNode &son2 = mem->son(*cluster2, j);
168  sons_(i, j).parameters_ = parameters_;
169  sons_(i, j).cluster1_ = std::addressof(son1);
170  sons_(i, j).cluster2_ = std::addressof(son2);
171  sons_(i, j).rows_ =
172  std::distance(mem->cluster_begin(son1), mem->cluster_end(son1));
173  sons_(i, j).cols_ =
174  std::distance(mem->cluster_begin(son2), mem->cluster_end(son2));
175  // let recursion handle the rest
176  sons_(i, j).appendSubtree(linear_operator, ansatz_space,
177  std::addressof(son1), std::addressof(son2));
178  }
179  } else {
180  leaf_ = std::make_shared<
181  TreeLeaf<Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>>>();
182  }
183  return;
184  }

255  private:
259  int compareCluster(const ElementTreeNode &cluster1,
260  const ElementTreeNode &cluster2) {
261  int r;
262  double dist = (cluster1.midpoint_ - cluster2.midpoint_).norm() -
263  cluster1.radius_ - cluster2.radius_;
264  dist = dist > 0 ? dist : 0;
265  double max_radius = cluster1.radius_ > cluster2.radius_ ? cluster1.radius_
266  : cluster2.radius_;
267 
271  if (dist < 0 || max_radius >= parameters_->eta_ * dist)
272  r = BlockClusterAdmissibility::Refine;
273  else
274  r = BlockClusterAdmissibility::LowRank;
275 
276  if (r == BlockClusterAdmissibility::Refine &&
277  parameters_->max_level_ - cluster1.level_ <=
278  parameters_->min_cluster_level_)
279  r = BlockClusterAdmissibility::Dense;
280 
281  return r;
282  }
283  void updateLeafPointers() {
284  // make sure we have the root of the block cluster tree calling this method
285  if (cluster1_->id_ != -1 || cluster1_ != cluster2_) {
286  return;
287  } else {
288  leaf_pointers_->clear();
289  updateLeafPointers_recursion(*this);
290  }
291  }
292  void updateLeafPointers_recursion(BlockClusterTree &child) {
293  if (child.cc_ == BlockClusterAdmissibility::Refine) {
294  for (auto j = 0; j < child.sons_.cols(); ++j)
295  for (auto i = 0; i < child.sons_.rows(); ++i)
296  updateLeafPointers_recursion(child.sons_(i, j));
297  } else
298  leaf_pointers_->push_back(std::addressof(child));
299  return;
300  }

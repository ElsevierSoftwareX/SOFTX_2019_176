87  Eigen::SparseMatrix<double> assembleGlueMatrix(
88  const SuperSpace<Derived> &superspace, const Projector<Derived> &proj) {
89  const int pre_dofs = proj.get_dofs_after_projector();
90  // The dofs that need to be identified with each other are divided into
91  // master and slaves, where the master is the firs w.r.t. the tp-ordering
92  std::vector<Eigen::Triplet<double>> trips;
93  std::vector<GlueRoutines::dofIdentification> dof_id =
94  makeDofIdentificationList(superspace, proj);
95 
96  // We keep track of certain information
97  std::vector<bool> dof_is_slave(pre_dofs, false);
98  std::vector<bool> dof_is_master(pre_dofs, false);
99 
100  // Here, we fill the two vectors above
101  int number_of_slaves = 0;
102  for (auto dofset : dof_id) {
103  // This sorting is required, since by construction only dofs[0]<dofs[1] is
104  // given, but in the case of corners and H1 discretizations, it might
105  // happen that dofs[0]>dofs[2]. Moreover, we count how many dofs we "glue
106  // away".
107  std::sort(dofset.dofs.begin(), dofset.dofs.end());
108  dof_is_master[dofset.dofs[0]] = true;
109  for (int i = 1; i < dofset.dofs.size(); ++i) {
110  dof_is_slave[dofset.dofs[i]] = true;
111  ++number_of_slaves;
112  }
113  }
114 
115  // Now, we sort w.r.t. the masters.
116  std::sort(dof_id.begin(), dof_id.end(),
117  [](GlueRoutines::dofIdentification a,
118  GlueRoutines::dofIdentification b) {
119  return a.dofs[0] < b.dofs[0];
120  });
121 
122  const int post_dofs = pre_dofs - number_of_slaves;
123 
124  // This block is the heart of the algorithms. Skip keeps track of how many
125  // slaves have been skipped already, and master_index keeps track on which
126  // master is about to be glued next.
127  int skip = 0;
128  int master_index = 0;
129  for (int i = 0; i < post_dofs; ++i) {
130  const int post_index = i;
131  while (dof_is_slave[i + skip] && ((i + skip) < pre_dofs)) ++skip;
132  const int pre_index = i + skip;
133  trips.push_back(Eigen::Triplet<double>(pre_index, post_index, 1));
134  // The dof cannot be declared slave and master at the same time
135  assert(not(dof_is_master[pre_index] && dof_is_slave[pre_index]));
136  if (dof_is_master[pre_index]) {
137  // The dofs in dof_id dont know they might be moved forward. So the
138  // smalles dof of those to be identified should coinicde with the
139  // pre_index.
140  assert(pre_index == dof_id[master_index].dofs[0]);
141  const int number_of_partners = dof_id[master_index].dofs.size();
142  for (int j = 1; j < number_of_partners; ++j) {
143  trips.push_back(Eigen::Triplet<double>(dof_id[master_index].dofs[j],
144  post_index,
145  dof_id[master_index].coef));
146  }
147  ++master_index;
148  }
149  }
150 
151  Eigen::SparseMatrix<double> glue_matrix(pre_dofs, post_dofs);
152  glue_matrix.setFromTriplets(trips.begin(), trips.end());
153 
154  assert(trips.size() == pre_dofs);
155  return glue_matrix;
156  }

161 // This routine collects the inices of DOFs with support on an edge in
162 // dependence of the dimension of the space in each tensor product direction
163 // and the edge. The shift parameter can be used to shift it to the correct
164 // patch or vector component.
165 std::vector<int> getEdgeDofIndices(int edgeCase, int dimXdir, int dimYdir,
166  int shift) {
167  std::vector<int> out;
168  switch (edgeCase) {
169  case (0): {
170  out.reserve(dimXdir);
171  assert(
172  "This should be a given. If not something went wrong in Glue.hpp" &&
173  dimXdir <= dimYdir);
174  for (int i = 0; i < dimXdir; ++i) {
175  out.push_back(i + shift);
176  }
177  return out;
178  };
179  case (1): {
180  out.reserve(dimYdir);
181  assert(
182  "This should be a given. If not something went wrong in Glue.hpp" &&
183  dimYdir <= dimXdir);
184  for (int i = 0; i < dimYdir; ++i) {
185  out.push_back(dimXdir * (i + 1) - 1 + shift);
186  }
187  return out;
188  };
189  case (2): {
190  out.reserve(dimXdir);
191  assert(
192  "This should be a given. If not something went wrong in Glue.hpp" &&
193  dimXdir <= dimYdir);
194  for (int i = 0; i < dimXdir; ++i) {
195  out.push_back(dimXdir * (dimYdir - 1) + i + shift);
196  }
197  return out;
198  };
199  case (3): {
200  out.reserve(dimYdir);
201  assert(
202  "This should be a given. If not something went wrong in Glue.hpp" &&
203  dimYdir <= dimXdir);
204  for (int i = 0; i < dimYdir; ++i) {
205  out.push_back(dimXdir * i + shift);
206  }
207  return out;
208  };
209  default:;
210  // An edge might have a -1 index. This occurs only when no partner could
211  // be found, and the -1 is the placeholder of the missing partner.
212  }
213  return out;
214 }

216 // The following 7 functions figure out how edges and the vector components
217 // are oriented w.r.t. each other.
218 inline bool edgeIsForwardParametrized(int edgeCase) {
219  return edgeCase == 0 || edgeCase == 1;
220 }
221 inline bool edgeIsBackwardsParametrized(int edgeCase) {
222  return not(edgeIsForwardParametrized(edgeCase));
223 }
224 inline bool normalComponentIsInwardDirected(int edgeCase) {
225  return edgeCase == 0 || edgeCase == 3;
226 }
227 inline bool normalComponentIsOutwardDirected(int edgeCase) {
228  return not(normalComponentIsInwardDirected(edgeCase));
229 }

231 inline bool reverseParametrized(const std::array<int, 4> &edge) {
232  return ((edgeIsForwardParametrized(edge[2]) &&
233  edgeIsForwardParametrized(edge[3])) ||
234  (edgeIsBackwardsParametrized(edge[2]) &&
235  edgeIsBackwardsParametrized(edge[3])))
236  ? true
237  : false;
238 };

240 inline int glueCoefficientDivergenceConforming(const std::array<int, 4> &edge) {
241  return ((normalComponentIsInwardDirected(edge[2]) &&
242  normalComponentIsInwardDirected(edge[3])) ||
243  (normalComponentIsOutwardDirected(edge[2]) &&
244  normalComponentIsOutwardDirected(edge[3])))
245  ? -1
246  : 1;

273 template <typename Derived>
274 struct glue_identificationmaker_<Derived, DifferentialForm::Continuous> {
275  static std::vector<dofIdentification> makeIdentification(
276  const std::vector<std::array<int, 4>> &edges_,
277  const SuperSpace<Derived> &superspace, const Projector<Derived> &proj) {
278  // We check if the space can even be continuous globally.
279  assert(superspace.get_polynomial_degree() >= 1);
280  const int one_d_dim = superspace.get_polynomial_degree() + 1 +
281  proj.get_knot_repetition() *
282  ((1 << superspace.get_refinement_level()) - 1);
283  const int dofs_per_patch = one_d_dim * one_d_dim;
284 
285  std::vector<GlueRoutines::dofIdentification> out;
286  out.reserve(edges_.size() * one_d_dim);
287 
288  std::vector<int> already_stored_in(proj.get_dofs_after_projector(), -1);
289  int d_count = 0;
290  for (auto edge : edges_) {
291  // The shift is to account for different patches, since the
292  // getEdgeDof-routines only can enumerate w.r.t. patch 0.
293  const int shift_e1 = edge[0] * dofs_per_patch;
294  const int shift_e2 = edge[1] * dofs_per_patch;
295  const bool needReversion = reverseParametrized(edge);
296 
297  std::vector<int> dofs_e1 =
298  getEdgeDofIndices(edge[2], one_d_dim, one_d_dim, shift_e1);
299  std::vector<int> dofs_e2 =
300  getEdgeDofIndices(edge[3], one_d_dim, one_d_dim, shift_e2);
301 
302  // Check if the edge is hanging. For screens one would need an "else".
303  if (edge[1] > -1 && edge[0] > -1) {
304  for (int i = 0; i < one_d_dim; ++i) {
305  GlueRoutines::dofIdentification d;
306  const int j = needReversion ? one_d_dim - 1 - i : i;
307  // const int j = i;
308  assert(dofs_e1[i] != dofs_e2[j] &&
309  "If this happens something went horribly wrong.");
310 
311  const int small_dof = std::min(dofs_e1[i], dofs_e2[j]);
312  const int large_dof = std::max(dofs_e1[i], dofs_e2[j]);
313 
314  // In the continuous case, more than two dofs might be glued together.
315  // Therefore, we must check if we know one of the dofs already. If
316  // yes, we just add the new one to the dof_id, if not, we make a new
317  // identification group.
318  if (already_stored_in[small_dof] > -1) {
319  assert(already_stored_in[large_dof] == -1);
320  out[already_stored_in[small_dof]].dofs.push_back(large_dof);
321  already_stored_in[large_dof] = already_stored_in[small_dof];
322  } else if (already_stored_in[large_dof] > -1) {
323  assert(already_stored_in[small_dof] == -1);
324  out[already_stored_in[large_dof]].dofs.push_back(small_dof);
325  already_stored_in[small_dof] = already_stored_in[large_dof];
326  } else {
327  d.dofs.push_back(small_dof);
328  d.dofs.push_back(large_dof);
329  already_stored_in[small_dof] = d_count;
330  already_stored_in[large_dof] = d_count;
331  d_count++;
332  d.coef = 1;
333  out.push_back(d);
334  }
335  }
336  }
337  }
338 
339  return out;
340  }
341 };

349 template <typename Derived>
350 struct glue_identificationmaker_<Derived, DifferentialForm::DivConforming> {
351  static std::vector<dofIdentification> makeIdentification(
352  const std::vector<std::array<int, 4>> &edges_,
353  const SuperSpace<Derived> &superspace, const Projector<Derived> &proj) {
354  // since we assume px == py and uniform refinement in both directions, there
355  // will be a small_dim and a large_dim in every vector component.
356  const int small_dim = superspace.get_polynomial_degree() +
357  proj.get_knot_repetition() *
358  ((1 << superspace.get_refinement_level()) - 1);
359  const int large_dim = superspace.get_polynomial_degree() + 1 +
360  proj.get_knot_repetition() *
361  ((1 << superspace.get_refinement_level()) - 1);
362  const int dofs_per_patch_per_component = small_dim * large_dim;
363  // sanity check
364  assert(dofs_per_patch_per_component ==
365  (proj.get_dofs_after_projector() /
366  (superspace.get_number_of_patches() * 2)) &&
367  "The assembly of the glue matrix is highly specific to the space. "
368  "Something went wrong; is the discrete space correct?");
369 
370  std::vector<GlueRoutines::dofIdentification> out;
371  out.reserve(edges_.size() * small_dim);
372 
373  for (auto edge : edges_) {
374  assert(edge[0] <= edge[1] || edge[1] == -1);
375  // The DOFs that need to be glued on edges 0 and 2 are in the second
376  // vector component. The remainder of the dof-index-shift is to account
377  // for dofs on other patches.
378  const int shift_e1 = edge[0] * dofs_per_patch_per_component +
379  (edgeToBeGluedInFirstComp(edge[2])
380  ? 0
381  : (dofs_per_patch_per_component *
382  superspace.get_number_of_patches()));
383  const int shift_e2 = edge[1] * dofs_per_patch_per_component +
384  (edgeToBeGluedInFirstComp(edge[3])
385  ? 0
386  : (dofs_per_patch_per_component *
387  superspace.get_number_of_patches()));
388 
389  const int xdir_dim_1 =
390  edgeToBeGluedInFirstComp(edge[2]) ? large_dim : small_dim;
391  const int ydir_dim_1 =
392  edgeToBeGluedInFirstComp(edge[2]) ? small_dim : large_dim;
393  const int xdir_dim_2 =
394  edgeToBeGluedInFirstComp(edge[3]) ? large_dim : small_dim;
395  const int ydir_dim_2 =
396  edgeToBeGluedInFirstComp(edge[3]) ? small_dim : large_dim;
397  std::vector<int> dofs_e1 =
398  getEdgeDofIndices(edge[2], xdir_dim_1, ydir_dim_1, shift_e1);
399  std::vector<int> dofs_e2 =
400  getEdgeDofIndices(edge[3], xdir_dim_2, ydir_dim_2, shift_e2);
401 
402  const int size_of_edge_dofs = dofs_e1.size();
403  assert(size_of_edge_dofs == dofs_e2.size() || edge[1] == -1);
404 
405  const bool needReversion = reverseParametrized(edge);
406  const int coef = glueCoefficientDivergenceConforming(edge);
407 
408  // Check if the edge is hanging. For screens one would need an "else".
409  if (edge[1] > -1 && edge[0] > -1) {
410  for (int i = 0; i < size_of_edge_dofs; ++i) {
411  GlueRoutines::dofIdentification d;
412  const int j = needReversion ? size_of_edge_dofs - 1 - i : i;
413  assert(dofs_e1[i] != dofs_e2[j] &&
414  "If this happens something went horribly wrong.");
415  d.dofs.push_back(std::min(dofs_e1[i], dofs_e2[j]));
416  d.dofs.push_back(std::max(dofs_e1[i], dofs_e2[j]));
417  d.coef = coef;
418  out.push_back(d);
419  }
420  }
421  }
422  out.shrink_to_fit();
423  return out;
424  }

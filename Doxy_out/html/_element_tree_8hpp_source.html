33  void init_ElementTree(const Geometry &g, unsigned int max_level) {
34  // initialise the data fields
35  geometry_ = g.get_geometry_ptr();
36  mem_ = std::make_shared<ElementTreeMemory>();
37  mem_->number_of_patches_ = geometry_->size();
38  mem_->max_level_ = max_level;
39  max_level_ = max_level;
40  number_of_patches_ = mem_->number_of_patches_;
41  // balanced quadTree > the number of elements per branch is
42  // \sum_{l=0}^L 4^l = (4^{L+1}-1) / 3;
43  mem_->memory_ = std::make_shared<std::vector<ElementTreeNode>>();
44  mem_->memory_->resize(mem_->cumNumElements(max_level));
45  // create the patches and set up the topology
46  {
47  std::vector<Eigen::Vector3d> uniquePts;
48  std::vector<ElementTreeNode *> patches;
49  Eigen::Vector3d v;
50  number_of_points_ = 0;
51  // set hold all element
52  ElementTreeNode &root = mem_->get_root();
53  root.sons_.resize(mem_->number_of_patches_);
54  root.set_memory(mem_);
55 
56  for (auto i = 0; i < number_of_patches_; ++i) {
57  root.sons_[i] = i + 1;
58  mem_->son(root, i).adjcents_ = std::vector<int>(4, -1);
59  mem_->son(root, i).vertices_.resize(4);
60  mem_->son(root, i).id_ = i;
61  mem_->son(root, i).level_ = 0;
62  mem_->son(root, i).patch_ = i;
63  mem_->son(root, i).set_memory(mem_);
64  // get images of the four corners of the unit square under the
65  // diffeomorphism geo[i]
66  for (auto j = 0; j < 4; ++j) {
67  v = (*geometry_)[i].eval(Constants::corners[0][j],
68  Constants::corners[1][j]);
69  unsigned int index = 0;
70  for (; index < uniquePts.size(); ++index)
71  if ((uniquePts[index] - v).norm() < Constants::pt_comp_tolerance)
72  break;
73  if (index != uniquePts.size())
74  mem_->son(root, i).vertices_[j] = index;
75  else {
76  uniquePts.push_back(v);
77  mem_->son(root, i).vertices_[j] = number_of_points_;
78  ++number_of_points_;
79  }
80  }
81  patches.push_back(std::addressof(mem_->son(root, i)));
82  }
83  updateTopology(patches);
84  }
85  // refine mesh
86  {
87  for (auto level = 0; level < max_level_; ++level) {
88  auto offset1 = mem_->cumNumElements(level - 1);
89  auto offset2 = mem_->cumNumElements(level);
90  for (auto j = offset1; j < offset2; ++j) refineLeaf(j);
91  }
92  }
93  return;
94  }

248  static void computeEnclosingBall(Eigen::Vector3d *mp, double *r,
249  const Eigen::Vector3d &mp1, double r1,
250  const Eigen::Vector3d &mp2, double r2) {
251  // compute distance vector of the two spheres
252  auto z = mp1 - mp2;
253  auto norm = (mp1 - mp2).norm();
255  if (norm + r2 <= r1) {
256  *mp = mp1;
257  *r = r1;
259  } else if (norm + r1 <= r2) {
260  *mp = mp2;
261  *r = r2;
263  } else {
264  *mp = 0.5 * (mp1 + mp2 + (r1 - r2) / norm * z);
265  *r = 0.5 * (r1 + r2 + norm);
266  *r = 0.5 * (r1 + r2 + norm);
267  }
268  return;
269  }
273  std::vector<std::array<int, 4>> patchTopologyInfo() const {
274  std::vector<std::array<int, 4>> retval;
275  for (auto it = mem_->clbegin(0); it != mem_->clend(0); ++it) {
276  for (auto j = 0; j < 4; ++j) {
277  // do we have a neighbour?
278  if (it->adjcents_[j] != -1) {
279  const ElementTreeNode &cur_neighbour = mem_->adjcent(*it, j);
280  // add the edge only if it->id_ < neighbour->id_
281  // otherwise the edge has already been added
282  if (it->id_ < cur_neighbour.id_) {
283  int k = 0;
284  for (; k < 4; ++k)
285  if (cur_neighbour.adjcents_[k] != -1 &&
286  mem_->adjcent(cur_neighbour, k).id_ == it->id_)
287  break;
288  retval.push_back({it->id_, cur_neighbour.id_, j, k});
289  }
290  } else
291  retval.push_back({it->id_, -1, j, -1});
292  }
293  }
294  return retval;
295  }

299  std::vector<int> computeReorderingVector() const {
300  std::vector<int> out(get_number_of_elements());
301  for (auto it = mem_->cpbegin(); it != mem_->cpend(); ++it) {
302  const double h = it->get_h();
303  const Eigen::Vector2d ref_mid =
304  it->llc_ + Eigen::Vector2d(.5 * h, .5 * h);
305  const int x_idx = std::floor(ref_mid(0) / h);
306  const int y_idx = std::floor(ref_mid(1) / h);
307  const int num_in_one_dir = 1 << max_level_;
308  const int num_per_patch = num_in_one_dir * num_in_one_dir;
309  const int tp_idx =
310  it->patch_ * num_per_patch + y_idx * num_in_one_dir + x_idx;
311  out[tp_idx] = it->id_;
312  }
313  return out;

350  void updateTopology(const std::vector<ElementTreeNode *> &elements) {
351  std::set<std::array<int, 3>, edgeComp<std::array<int, 3>>> edges;
352  std::array<int, 3> e1, e2;
353  // generate edge list for all elements in question
354  for (auto i = 0; i < elements.size(); ++i) {
355  // iterate over the four edges of each element
356  for (auto j = 0; j < 4; ++j) {
357  // compute a unique id for each edge
358  if (elements[i]->vertices_[j] < elements[i]->vertices_[(j + 1) % 4]) {
359  e1[0] = elements[i]->vertices_[j];
360  e1[1] = elements[i]->vertices_[(j + 1) % 4];
361  } else {
362  e1[1] = elements[i]->vertices_[j];
363  e1[0] = elements[i]->vertices_[(j + 1) % 4];
364  }
365  e1[2] = i;
366  // perferm a look up if the edge is already existing.
367  auto it = edges.find(e1);
368  // if so, identify the two neighbours
369  if (it != edges.end()) {
370  elements[i]->adjcents_[j] =
371  mem_->cumNumElements(elements[(*it)[2]]->level_ - 1) +
372  elements[(*it)[2]]->id_;
373  // now find the edge also in the patch that added it to the set
374  for (auto k = 0; k < 4; ++k) {
375  if (elements[(*it)[2]]->vertices_[k] <
376  elements[(*it)[2]]->vertices_[(k + 1) % 4]) {
377  e2[0] = elements[(*it)[2]]->vertices_[k];
378  e2[1] = elements[(*it)[2]]->vertices_[(k + 1) % 4];
379  } else {
380  e2[1] = elements[(*it)[2]]->vertices_[k];
381  e2[0] = elements[(*it)[2]]->vertices_[(k + 1) % 4];
382  }
383  if (e1[0] == e2[0] && e1[1] == e2[1])
384  elements[(*it)[2]]->adjcents_[k] =
385  mem_->cumNumElements(elements[i]->level_ - 1) +
386  elements[i]->id_;
387  }
388  // otherwise add the edge to the list
389  } else
390  edges.insert(e1);
391  }
392  }
393  return;

395  void refineLeaf(int global_id) {
396  // check if we have actually a panel
397  ElementTreeNode &cur_el = mem_->get_element(global_id);
398  if (cur_el.sons_.size()) return;
399  std::vector<int> ptIds(5);
400  std::vector<int> refNeighbours(4);
401  std::vector<ElementTreeNode *> elements;
402  // determine position of p with respect to its neighbours
403  for (auto i = 0; i < 4; ++i) {
404  // is there a neighbour?
405  if (cur_el.adjcents_[i] != -1) {
406  for (auto j = 0; j < 4; ++j)
407  if (mem_->adjcent(cur_el, i).adjcents_[j] == global_id) {
408  refNeighbours[i] = j;
409  break;
410  }
411  } else
412  refNeighbours[i] = -1;
413  }
414  // determine new points
415  for (auto i = 0; i < 4; ++i) {
416  auto cur_neighbour = cur_el.adjcents_[i];
417  ElementTreeNode &ref_cur_neighbour = mem_->adjcent(cur_el, i);
418  // is the neighbour already refined?
419  if (cur_neighbour != -1 && ref_cur_neighbour.sons_.size()) {
420  // this is the midpoint of the shared edge
421  ptIds[i] = mem_->son(ref_cur_neighbour, refNeighbours[i])
422  .vertices_[(refNeighbours[i] + 1) % 4];
423  // these are the two elements adjacent to the edge
424  elements.push_back(
425  std::addressof(mem_->son(ref_cur_neighbour, refNeighbours[i])));
426  elements.push_back(std::addressof(
427  mem_->son(ref_cur_neighbour, (refNeighbours[i] + 1) % 4)));
428  }
429  // otherwise add the point id to the tree
430  else {
431  ptIds[i] = number_of_points_;
432  ++number_of_points_;
433  }
434  }
435  // add midpoint of the current element, which is always a new point
436  ptIds[4] = number_of_points_;
437  ++number_of_points_;
438  // set up new elements
439  cur_el.sons_.resize(4);
440  for (auto i = 0; i < 4; ++i) {
441  auto child_ind = 4 * cur_el.id_ + i + mem_->cumNumElements(cur_el.level_);
442  cur_el.sons_[i] = child_ind;
443  mem_->son(cur_el, i).patch_ = cur_el.patch_;
444  mem_->son(cur_el, i).level_ = cur_el.level_ + 1;
445  mem_->son(cur_el, i).id_ = 4 * cur_el.id_ + i;
446  mem_->son(cur_el, i).adjcents_ = std::vector<int>(4, -1);
447  mem_->son(cur_el, i).llc_(0) =
448  cur_el.llc_(0) + Constants::llcs[0][i] / (1 << cur_el.level_);
449  mem_->son(cur_el, i).llc_(1) =
450  cur_el.llc_(1) + Constants::llcs[1][i] / (1 << cur_el.level_);
451  mem_->son(cur_el, i).set_memory(mem_);
452  elements.push_back(std::addressof(mem_->son(cur_el, i)));
453  }
454  // set vertices
455  mem_->son(cur_el, 0).vertices_.push_back(cur_el.vertices_[0]);
456  mem_->son(cur_el, 0).vertices_.push_back(ptIds[0]);
457  mem_->son(cur_el, 0).vertices_.push_back(ptIds[4]);
458  mem_->son(cur_el, 0).vertices_.push_back(ptIds[3]);
459 
460  mem_->son(cur_el, 1).vertices_.push_back(ptIds[0]);
461  mem_->son(cur_el, 1).vertices_.push_back(cur_el.vertices_[1]);
462  mem_->son(cur_el, 1).vertices_.push_back(ptIds[1]);
463  mem_->son(cur_el, 1).vertices_.push_back(ptIds[4]);
464 
465  mem_->son(cur_el, 2).vertices_.push_back(ptIds[4]);
466  mem_->son(cur_el, 2).vertices_.push_back(ptIds[1]);
467  mem_->son(cur_el, 2).vertices_.push_back(cur_el.vertices_[2]);
468  mem_->son(cur_el, 2).vertices_.push_back(ptIds[2]);
469 
470  mem_->son(cur_el, 3).vertices_.push_back(ptIds[3]);
471  mem_->son(cur_el, 3).vertices_.push_back(ptIds[4]);
472  mem_->son(cur_el, 3).vertices_.push_back(ptIds[2]);
473  mem_->son(cur_el, 3).vertices_.push_back(cur_el.vertices_[3]);
474 
475  updateTopology(elements);
476 
477  return;
478  }

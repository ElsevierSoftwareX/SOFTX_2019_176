23  int next = 0;
24  // assume isotropic mesh width h!

26  auto nE = super_space.get_mesh().get_number_of_elements();
27  auto pbegin = super_space.get_mesh().get_element_tree().cpbegin();
28  auto pend = super_space.get_mesh().get_element_tree().cpend();
29  ffield_qnodes.resize(12, nE * Q.xi_.cols());
30  SurfacePoint surfpt;
31 
32  for (auto it = pbegin; it != pend; ++it)
33  for (auto k = 0; k < Q.xi_.cols(); ++k) {
34  // the quadrature weight is scaled by mesh width
35  // this corresponds to a scaling of the basis functions
36  // with respect to the L^2 norm!
37  super_space.map2surface(*it, Q.xi_.col(k), h * Q.w_(k), &surfpt);
38  ffield_qnodes.col(it->id_ * Q.xi_.cols() + k) = surfpt;
39  }
40  return ffield_qnodes;
41 }
